import torch
import pyro
import pyro.distributions as dist
import gpytorch as gp
import matplotlib.pyplot as plt

def generate_data_with_fixed_scales(num_individuals, num_diseases, num_time_points, 
                                    num_covariates, K, g_i, mu_d, 
                                    fixed_length_scales_lambda, fixed_variances_lambda, 
                                    fixed_length_scales_phi, fixed_variances_phi):
    T = num_time_points
    N = num_individuals
    D = num_diseases
    
    time_points = torch.arange(T, dtype=torch.float32).unsqueeze(1)
    
    # Simulate gamma from a multivariate normal
    gamma_loc = torch.zeros(K, num_covariates, dtype=torch.float32)
    gamma_cov = torch.eye(num_covariates, dtype=torch.float32)
    gamma = dist.MultivariateNormal(gamma_loc, gamma_cov).sample()

    # Initialize tensor for Lambda (N x K x T) and Phi (K x D x T)
    lambda_ik = torch.zeros(N, K, T, dtype=torch.float32)
    phi_kd = torch.zeros(K, D, T, dtype=torch.float32)

    # Simulate Lambda (N x K x T) using fixed length scales and variance scales
    for k in range(K):
        kernel = gp.kernels.RBFKernel()
        kernel.lengthscale = fixed_length_scales_lambda[k]
        cov_matrix_lambda = kernel(time_points).evaluate() * fixed_variances_lambda[k] + 1e-5 * torch.eye(T)
        
        for n in range(N):
            lambda_loc = torch.matmul(gamma[k], g_i[n]).expand(T)
            lambda_ik[n, k, :] = dist.MultivariateNormal(loc=lambda_loc, covariance_matrix=cov_matrix_lambda).sample()

    # Simulate Phi (K x D x T) using fixed length scales and variance scales
    for k in range(K):
        kernel = gp.kernels.RBFKernel()
        kernel.lengthscale = fixed_length_scales_phi[k]
        cov_matrix_phi = kernel(time_points).evaluate() * fixed_variances_phi[k] + 1e-5 * torch.eye(T)
        
        for d in range(D):
            phi_loc = torch.zeros(T, dtype=torch.float32)
            phi_kd[k, d, :] = dist.MultivariateNormal(loc=phi_loc, covariance_matrix=cov_matrix_phi).sample()

    # Calculate eta (N x D x T) using einsum and the fixed mu_d
    eta = mu_d.unsqueeze(0) + torch.einsum('nkt,kdt->ndt', lambda_ik, phi_kd)

    # Convert eta to probabilities pi (N x D x T)
    pi = torch.sigmoid(eta)

    # Initialize a mask to prevent re-diagnosis
    diagnosis_mask = torch.zeros(N, D, dtype=torch.float32)

    # Sample data and enforce no re-diagnosis constraint
    data = torch.zeros(N, D, T, dtype=torch.float32)
    for t in range(T):
        pi[:, :, t] *= (1 - diagnosis_mask)
        new_diagnoses = dist.Bernoulli(pi[:, :, t]).sample()
        data[:, :, t] = new_diagnoses
        diagnosis_mask = torch.max(diagnosis_mask, new_diagnoses)

    return data, pi, gamma, lambda_ik, phi_kd

def aladynoulli_model_fixed(data, g_i, mu_d, num_covariates, num_topics,
                            fixed_length_scales_lambda, fixed_variances_lambda,
                            fixed_length_scales_phi, fixed_variances_phi):
    N, D, T = data.shape
    K = num_topics
    
    time_points = torch.arange(T, dtype=torch.float32).unsqueeze(1)

    # Sample Gamma (global influence across topics and covariates)
    gamma_loc = torch.zeros(K, num_covariates)
    gamma_cov = torch.eye(num_covariates)
    gamma = pyro.sample("gamma", dist.MultivariateNormal(gamma_loc, gamma_cov).to_event(1))

    # Sample Lambda (topic weights) using fixed length scales and variances
    lambda_ik = torch.zeros(N, K, T)
    for k in range(K):
        kernel = gp.kernels.RBFKernel()
        kernel.lengthscale = fixed_length_scales_lambda[k]
        cov_matrix_lambda = kernel(time_points).evaluate() * fixed_variances_lambda[k] + 1e-5 * torch.eye(T)
        
        for n in range(N):
            lambda_loc = torch.matmul(g_i[n], gamma[k]).expand(T)
            lambda_ik[n, k, :] = pyro.sample(f"lambda_{k}_{n}", dist.MultivariateNormal(lambda_loc, cov_matrix_lambda))

    # Sample Phi (disease-topic loadings) using fixed length scales and variances
    phi_kd = torch.zeros(K, D, T)
    for k in range(K):
        kernel = gp.kernels.RBFKernel()
        kernel.lengthscale = fixed_length_scales_phi[k]
        cov_matrix_phi = kernel(time_points).evaluate() * fixed_variances_phi[k] + 1e-5 * torch.eye(T)
        
        for d in range(D):
            phi_kd[k, d, :] = pyro.sample(f"phi_{k}_{d}", dist.MultivariateNormal(torch.zeros(T), cov_matrix_phi))

    # Calculate eta and pi
    eta = mu_d.unsqueeze(0) + torch.einsum('nkt,kdt->ndt', lambda_ik, phi_kd)
    pi = torch.sigmoid(eta)

    # Observe data with no re-diagnosis constraint
    diagnosis_mask = torch.zeros(N, D)
    
    for t in range(T):
        pi_t = pi[:, :, t] * (1 - diagnosis_mask)
        
        # Observe data
        pyro.sample(f"obs_{t}", dist.Bernoulli(pi_t).to_event(2), obs=data[:, :, t])
        
        # Update diagnosis mask
        diagnosis_mask = torch.max(diagnosis_mask, data[:, :, t])

    return lambda_ik, phi_kd

def aladynoulli_guide(data, g_i, mu_d, num_covariates, num_topics):
    N, D, T = data.shape
    K = num_topics

    # Variational parameters for Gamma
    gamma_loc = pyro.param("gamma_loc", torch.zeros((K, num_covariates)))
    gamma_scale = pyro.param("gamma_scale", torch.ones((K, num_covariates)), constraint=dist.constraints.positive)
    pyro.sample("gamma", dist.Normal(gamma_loc, gamma_scale).to_event(2))

    # Variational parameters for Lambda
    for k in range(K):
        for n in range(N):
            lambda_loc = pyro.param(f"lambda_{k}_{n}_loc", torch.zeros(T))
            lambda_scale = pyro.param(f"lambda_{k}_{n}_scale", torch.ones(T), constraint=dist.constraints.positive)
            pyro.sample(f"lambda_{k}_{n}", dist.Normal(lambda_loc, lambda_scale).to_event(1))

    # Variational parameters for Phi
    for k in range(K):
        for d in range(D):
            phi_loc = pyro.param(f"phi_{k}_{d}_loc", torch.zeros(T))
            phi_scale = pyro.param(f"phi_{k}_{d}_scale", torch.ones(T), constraint=dist.constraints.positive)
            pyro.sample(f"phi_{k}_{d}", dist.Normal(phi_loc, phi_scale).to_event(1))

def initialize_model(data, g_i, mu_d, num_covariates, num_topics,
                     fixed_length_scales_lambda, fixed_variances_lambda,
                     fixed_length_scales_phi, fixed_variances_phi):
    pyro.clear_param_store()
    
    def model(data, g_i, mu_d, num_covariates, num_topics):
        return aladynoulli_model_fixed(data, g_i, mu_d, num_covariates, num_topics,
                                       fixed_length_scales_lambda, fixed_variances_lambda,
                                       fixed_length_scales_phi, fixed_variances_phi)
    
    def guide(data, g_i, mu_d, num_covariates, num_topics):
        return aladynoulli_guide(data, g_i, mu_d, num_covariates, num_topics)
    
    # Warm start
    N, D, T = data.shape
    K = num_topics

    # Initialize gamma
    pyro.param("gamma_loc", torch.randn((K, num_covariates)) * 0.1)
    pyro.param("gamma_scale", torch.ones((K, num_covariates)) * 0.1, constraint=dist.constraints.positive)

    # Initialize lambda
    for k in range(K):
        for n in range(N):
            pyro.param(f"lambda_{k}_{n}_loc", torch.randn(T) * 0.1)
            pyro.param(f"lambda_{k}_{n}_scale", torch.ones(T) * 0.1, constraint=dist.constraints.positive)

    # Initialize phi
    for k in range(K):
        for d in range(D):
            pyro.param(f"phi_{k}_{d}_loc", torch.randn(T) * 0.1)
            pyro.param(f"phi_{k}_{d}_scale", torch.ones(T) * 0.1, constraint=dist.constraints.positive)

    # Run the model and guide once to initialize parameters
    trace = pyro.poutine.trace(model).get_trace(data, g_i, mu_d, num_covariates, num_topics)
    trace = pyro.poutine.trace(guide).get_trace(data, g_i, mu_d, num_covariates, num_topics)
    
    return model, guide

def initialize_mu_d(num_diseases, num_time_points):
    return torch.randn(num_diseases, num_time_points) * 0.1
