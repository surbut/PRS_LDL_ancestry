# Set the number of components (topics)
K = 3  # adjust as needed

# Perform non-negative tensor factorization
factors = non_negative_parafac(Y_centered, rank=K, n_iter_max=1000, tol=1e-8)

phi_init = factors.factors[1]  # This is DxK
